\subsection{Exercises}

\begin{problem}\hfill\\
  Let $C$ denote the Cantor set.
\begin{enumerate}
  \item Show that $C$ contains point that is not an endpoint of one of the removed intervals.
  \item Show that $C$ is nowhere dense, meager, and has measure zero.
  \item Show that $C$ is uncountable.
\end{enumerate}
\end{problem}

\begin{solution}\hfill\\
  \begin{enumerate}
    \item First we will characterize the endpoints of the removed intervals. Let $C_n$ be the $n$th stage of the deleting process used to define the Cantor set; then
      $$
      C_n = \union [1, 2]
      $$
  \end{enumerate}
\end{solution}
