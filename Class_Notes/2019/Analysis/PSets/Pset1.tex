\subsection{Exercises}

\begin{problem}\hfill\\
  Let $C$ denote the Cantor set.
\begin{enumerate}
  \item Show that $C$ contains point that is not an endpoint of one of the removed intervals.
  \item Show that $C$ is nowhere dense, meager, and has measure zero.
  \item Show that $C$ is uncountable.
\end{enumerate}
\end{problem}

\begin{solution}\hfill\\
  \begin{enumerate}
    \item First we will characterize the endpoints of the removed intervals. Let $C_n$ be the $n$th stage of the deleting process used to define the Cantor set; then what remains is a union of intervals:
      $$
      C_n = [0, \frac 1 {3^n}] \union [\frac 2 {2^n}, \frac 3 {3^n}] \union \cdots [\frac {3^n-1} n, 1],
      $$
    and so the endpoints are precisely the numbers of the form $\frac{k}{3^n}$ where $1 \leq k \leq 3^n$.
  \end{enumerate}
\end{solution}
